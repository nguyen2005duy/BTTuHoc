 {

    {
        return false;
    }

    for (int i =0;i<sizea;i++)
    {
        for (int j = 0;j<sizeb;j++)
        {
            if (a[i]==b[i])
            {
                count++;
            }
        }
    }
    if (count==n)
    {
        return true;
    }
    return false;
